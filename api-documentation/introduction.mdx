```JavaScript
api.schedule(function(done) {
  performCalculations();

```

## Fetching a project's data

```JavaScript
  async function fetchProject() {
      try {
        const response = await axios({
          method: 'POST',
          headers: {
            'content-type': 'application/json'
          },
          data: {
            namespace: namespace,
            hostName: hostName
          },
          url: '/api/projects/find'
        })

        setProject(response.data)
      } catch (err) {
        console.error(err)
      }
    }
```
